● ver quantos caracteres tem uma string ● SELECT CHAR_LENGTH(nome_da_coluna) AS alias FROM nome_da_tabela; ● Concatenar strings ● SELECT CONCAT(coluna1, coluna2) AS alias FROM nome_da_tabela; ● Concatenar strings com argumento é um separador ● SELECT CONCAT_WS('-', coluna1, coluna2) AS alias FROM nome_da_tabela; ● Formatar strings ● SELECT FORMAT(coluna, casas_decimais) AS alias FROM nome_da_tabela; ● Localizar a posição da string ● SELECT INSTR(coluna, string) AS alias FROM nome_da_tabela; ● Converter para minusculo ● SELECT LCASE(coluna) AS alias FROM nome_da_tabela; ● Selecionar parte da string da esquerda ● SELECT LEFT(coluna, x) AS alias FROM nome_da_tabela; ● Selecionar parte da string da direita ● SELECT RIGHT(coluna, x) AS alias FROM nome_da_tabela; ● Substituir parte da string ● SELECT REPLACE(coluna, string1, string2) AS alias FROM nome_da_tabela; ● Selecionar parte da string (inicio(x) e fim (y)) ● SELECT SUBSTR(coluna, x, y) AS alias FROM nome_da_tabela; ● Converter para maiusculo ● SELECT UCASE(coluna) AS alias FROM nome_da_tabela;
