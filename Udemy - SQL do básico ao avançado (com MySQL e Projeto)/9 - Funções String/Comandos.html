● ver quantos caracteres tem uma string ● SELECT CHAR_LENGTH(nome_da_coluna) AS alias FROM nome_da_tabela; ● Concatenar strings ● SELECT CONCAT(coluna1, coluna2) AS alias FROM nome_da_tabela; ● Concatenar strings com argumento é um separador ● SELECT CONCAT_WS('-', coluna1, coluna2) AS alias FROM nome_da_tabela; ● Formatar strings ● SELECT FORMAT(coluna, casas_decimais) AS alias FROM nome_da_tabela; ● Localizar a posição da string ● SELECT INSTR(coluna, string) AS alias FROM nome_da_tabela;
